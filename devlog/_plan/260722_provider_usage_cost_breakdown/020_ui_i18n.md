## Scope and prerequisites

This phase is a UI-only continuation of Phase 1. It modifies exactly these files:

- `gui/src/pages/Providers.tsx`
- `gui/src/components/provider-workspace/ProviderDetails.tsx`
- `gui/src/components/provider-workspace/ProviderUsage.tsx`
- `gui/src/styles/provider-workspace-shell.css`
- `gui/src/i18n/en.ts`
- `gui/src/i18n/ko.ts`
- `gui/src/i18n/zh.ts`
- `gui/src/i18n/de.ts`
- `gui/src/i18n/ru.ts`

Phase 1 must already provide both of the following from
`gui/src/components/provider-workspace/types.ts` and
`gui/src/provider-workspace/usage.ts`:

```ts
export interface ProviderModelUsageRow {
  model: string;
  resolvedModel?: string;
  requests: number;
  totalTokens: number;
  inputTokens: number;
  outputTokens: number;
  shareRatio: number;
  estimatedCostUsd?: number;
}

export function formatCostUsd(
  value: number | null | undefined,
  locale = "en",
): string {
  if (value === null || value === undefined || !Number.isFinite(value) || value < 0) return "—";
  return `~$${new Intl.NumberFormat(locale, {
    minimumFractionDigits: 4,
    maximumFractionDigits: 4,
  }).format(value)}`;
}
```

`formatCostUsd` must use both `minimumFractionDigits: 4` and
`maximumFractionDigits: 4`, matching the existing Usage page. `ProviderDetails`
must also already pass `modelUsage` and the legacy `usageSummaryEstimatedCost`
through to `ProviderUsage` as Phase 1's compile-closure state. Phase 2 removes the
legacy summary prop from the explicit `Providers -> ProviderDetails ->
ProviderUsage` call path because provider cost now comes directly from each model
row. No API, pricing, or global Usage-page changes belong in this phase.

The provider cost KPI is the direct sum of the server-computed model costs:

```ts
modelUsage.reduce((sum, row) => sum + (row.estimatedCostUsd ?? 0), 0)
```

The calculation deliberately returns `undefined` when `modelUsage` itself was not
provided. This lets `formatCostUsd` render its unavailable marker instead of
misrepresenting missing model data as `$0`. A row whose server-provided cost is
undefined contributes `0` to the provider sum and renders the helper's unavailable
marker in its own cost cell.

## MODIFY — `gui/src/components/provider-workspace/ProviderUsage.tsx`

### Rationale

- Derive the provider cost by summing the server-provided `estimatedCostUsd`

  formatCostUsd,
  formatRelativeTime,
  relativeTimeLabelsFromT,
  formatRequestCount,
  formatTokenCount,
} from "../../provider-workspace/usage";
import { accountQuotaFromReport, formatQuotaSourceLabel, type ProviderQuotaReportView } from "../../provider-workspace/report";
import type { ProviderModelUsageRow, ProviderUsageTotals } from "./types";

export default function ProviderUsage({
  item,
  usageTotals,
  modelUsage,
  quotaReport,
}: {
  item: WorkspaceItem;
  usageTotals?: ProviderUsageTotals;
  modelUsage?: ProviderModelUsageRow[];
  quotaReport?: ProviderQuotaReportView;
}) {
  const t = useT();
  const { locale } = useI18n();
  const [expandedModel, setExpandedModel] = useState<string | null>(null);
  const timeLabels = relativeTimeLabelsFromT(t);

suppression must be removed. Rendering `modelUsage` removes its `void` suppression,
while deleting the legacy summary prop removes that suppression entirely. The
provider-qualified expansion key prevents an expanded model from appearing to remain
open when the detail pane switches to another provider with the same model name.

### Model breakdown insertion

Insert the following block after the first `.pws-usage-block` closes and before
the existing Rate limits `.pws-usage-block` begins.

Before:

```tsx
      </div>
      <div className="pws-usage-block">
        <h3 className="pws-section-title">{t("pws.rateLimits")}</h3>
```

After:

```tsx
      </div>
      <div className="pws-usage-block">
        <h3 className="pws-section-title">{t("pws.modelBreakdown")}</h3>
        {modelUsage?.length ? (
          <table className="pws-model-table" aria-label={t("pws.modelBreakdown")}>
            <thead>
              <tr>
                <th>{t("pws.col.model")}</th>
                <th className="num">{t("pws.col.cost")}</th>
                <th className="num">{t("pws.col.tokens")}</th>
                <th className="num">{t("pws.col.requests")}</th>
                <th>{t("pws.col.share")}</th>
              </tr>
            </thead>
            {modelUsage.map(row => {
              const modelName = row.resolvedModel ?? row.model;
              const modelKey = `${item.name}:${row.model}:${row.resolvedModel ?? ""}`;
              const isExpanded = expandedModel === modelKey;
              const sharePercent = Math.round(row.shareRatio * 100);
              return (
                <tbody key={modelKey}>
                  <tr>
                    <td className="mono">
                      <button
                        type="button"
                        className="pws-model-expand-btn"
                        aria-expanded={isExpanded}
                        aria-label={`${t("pws.modelBreakdown")}: ${modelName}`}
                        onClick={() => setExpandedModel(isExpanded ? null : modelKey)}
                      >
                        {isExpanded ? "▾" : "▸"}
                      </button>{" "}
                      {modelName}
                    </td>
                    <td className="num">{formatCostUsd(row.estimatedCostUsd, locale)}</td>
                    <td className="num">{formatTokenCount(row.totalTokens, locale)}</td>
                    <td className="num">{formatRequestCount(row.requests, locale)}</td>
                    <td>
                      <div
                        className="pws-share-bar"
                        role="progressbar"
                        aria-label={`${modelName} ${t("pws.col.share")}`}
                        aria-valuemin={0}
                        aria-valuemax={100}
                        aria-valuenow={sharePercent}
                      >
                        <div className="pws-share-bar-fill" style={{ width: `${sharePercent}%` }} />
                      </div>
                    </td>
                  </tr>
                  {isExpanded && (
                    <tr className="pws-model-detail">
                      <td colSpan={5}>
                        <div className="pws-model-detail-grid">
                          <div>
                            <span className="muted">{t("pws.tokenInput")}</span>{" "}
                            <span className="mono">{formatTokenCount(row.inputTokens, locale)}</span>
                          </div>
                          <div>
                            <span className="muted">{t("pws.tokenOutput")}</span>{" "}
                            <span className="mono">{formatTokenCount(row.outputTokens, locale)}</span>
                          </div>
                        </div>
                      </td>
                    </tr>
                  )}
                </tbody>
              );
            })}
          </table>
        ) : (
          <p className="muted">{t("pws.noModelUsage")}</p>
        )}
      </div>
      <div className="pws-usage-block">
        <h3 className="pws-section-title">{t("pws.rateLimits")}</h3>
```

Multiple keyed `<tbody>` elements are valid table structure and keep each summary
row paired with its optional detail row without adding a `Fragment` import. The
share bar intentionally mirrors the global Usage-page calculation; the API's
`shareRatio` contract is expected to remain in the inclusive `0..1` range.
The visible columns are Model, Est. cost, Tokens, Requests, and Share in that order;
the model cell remains monospace and the cost cell uses `formatCostUsd`.

  const t = useT();
  const { locale } = useI18n();
  const [expandedModel, setExpandedModel] = useState<string | null>(null);
  const timeLabels = relativeTimeLabelsFromT(t);
  const hasUsage = usageTotals?.requests !== undefined;
  const quota = accountQuotaFromReport(quotaReport);
  const providerEstimatedCost = useMemo(
    () => modelUsage?.reduce((sum, row) => sum + (row.estimatedCostUsd ?? 0), 0),
    [modelUsage],
  );

  return (
    <div className="pws-section">
      <div className="pws-usage-block">
        <h3 className="pws-section-title">{t("pws.usageLast30d")}</h3>
        {hasUsage ? (
          <>
            <div className="pws-usage-metrics pws-usage-metrics-3" role="group" aria-label={t("pws.usageLast30d")}>
              <div className="pws-usage-metric">
                <span className="pws-usage-metric-value">{formatCostUsd(providerEstimatedCost, locale)}</span>
                <span className="muted pws-usage-metric-label">{t("pws.estimatedCost")}</span>
              </div>
              <div className="pws-usage-metric">
                <span className="pws-usage-metric-value">{formatRequestCount(usageTotals?.requests, locale)}</span>
                <span className="muted pws-usage-metric-label">{t("pws.metricRequests")}</span>
              </div>
              <div className="pws-usage-metric">
                <span className="pws-usage-metric-value">{formatTokenCount(usageTotals?.totalTokens, locale)}</span>
                <span className="muted pws-usage-metric-label">{t("pws.metricTokens")}</span>
              </div>
            </div>
            <p className="muted pws-cost-disclaimer">{t("pws.costDisclaimer")}</p>
          </>
        ) : (
          <p className="muted">{t("pws.usageUnavailable")}</p>
        )}
      </div>
      <div className="pws-usage-block">
        <h3 className="pws-section-title">{t("pws.modelBreakdown")}</h3>
        {modelUsage?.length ? (
          <table className="pws-model-table" aria-label={t("pws.modelBreakdown")}>
            <thead>
              <tr>
                <th>{t("pws.col.model")}</th>
                <th className="num">{t("pws.col.cost")}</th>
                <th className="num">{t("pws.col.tokens")}</th>
                <th className="num">{t("pws.col.requests")}</th>
                <th>{t("pws.col.share")}</th>
              </tr>
            </thead>
            {modelUsage.map(row => {
              const modelName = row.resolvedModel ?? row.model;
              const modelKey = `${item.name}:${row.model}:${row.resolvedModel ?? ""}`;
              const isExpanded = expandedModel === modelKey;
              const sharePercent = Math.round(row.shareRatio * 100);
              return (
                <tbody key={modelKey}>
                  <tr>
                    <td className="mono">
                      <button
                        type="button"
                        className="pws-model-expand-btn"
                        aria-expanded={isExpanded}
                        aria-label={`${t("pws.modelBreakdown")}: ${modelName}`}
                        onClick={() => setExpandedModel(isExpanded ? null : modelKey)}
                      >
                        {isExpanded ? "▾" : "▸"}
                      </button>{" "}
                      {modelName}
                    </td>
                    <td className="num">{formatCostUsd(row.estimatedCostUsd, locale)}</td>
                    <td className="num">{formatTokenCount(row.totalTokens, locale)}</td>
                    <td className="num">{formatRequestCount(row.requests, locale)}</td>
                    <td>
                      <div
                        className="pws-share-bar"
                        role="progressbar"
                        aria-label={`${modelName} ${t("pws.col.share")}`}
                        aria-valuemin={0}
                        aria-valuemax={100}
                        aria-valuenow={sharePercent}
                      >
                        <div className="pws-share-bar-fill" style={{ width: `${sharePercent}%` }} />
                      </div>
                    </td>
                  </tr>
                  {isExpanded && (
                    <tr className="pws-model-detail">
                      <td colSpan={5}>
                        <div className="pws-model-detail-grid">
                          <div>
                            <span className="muted">{t("pws.tokenInput")}</span>{" "}
                            <span className="mono">{formatTokenCount(row.inputTokens, locale)}</span>
                          </div>
                          <div>
                            <span className="muted">{t("pws.tokenOutput")}</span>{" "}
                            <span className="mono">{formatTokenCount(row.outputTokens, locale)}</span>
                          </div>
                        </div>
                      </td>
                    </tr>
                  )}
                </tbody>
              );
            })}
          </table>
        ) : (
          <p className="muted">{t("pws.noModelUsage")}</p>
        )}
      </div>
      <div className="pws-usage-block">
        <h3 className="pws-section-title">{t("pws.rateLimits")}</h3>
        {quota ? (
          <>
            <QuotaBars quota={quota} plan={null} threshold={80} t={t} layout="stacked" />
            <dl className="pws-kv pws-usage-meta">
              {quotaReport?.source?.trim() && (
                <div className="pws-kv-row">
                  <dt>{t("pws.stats.source")}</dt>
                  <dd>{formatQuotaSourceLabel(quotaReport.source)}</dd>
                </div>
              )}
              <div className="pws-kv-row">
                <dt>{t("pws.stats.quotaUpdated")}</dt>
                <dd>{formatRelativeTime(quotaReport?.updatedAt, timeLabels)}</dd>
              </div>
            </dl>
          </>
        ) : (
          <p className="muted">{t("pws.quotaUnavailable")}</p>
        )}
      </div>
    </div>
  );
}
